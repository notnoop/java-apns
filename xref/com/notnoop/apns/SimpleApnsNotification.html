31  package com.notnoop.apns;
32  
33  import java.util.Arrays;
34  
35  import com.notnoop.apns.internal.Utilities;
36  
37  
38  /**
39   * Represents an APNS notification to be sent to Apple service.
40   */
41  public class SimpleApnsNotification implements ApnsNotification {
42  
43      private final static byte COMMAND = 0;
44      private final byte[] deviceToken;
45      private final byte[] payload;
46  
47      /**
48       * Constructs an instance of {@code ApnsNotification}.
49       *
50       * The message encodes the payload with a {@code UTF-8} encoding.
51       *
52       * @param dtoken    The Hex of the device token of the destination phone
53       * @param payload   The payload message to be sent
54       */
55      public SimpleApnsNotification(String dtoken, String payload) {
56          this.deviceToken = Utilities.decodeHex(dtoken);
57          this.payload = Utilities.toUTF8Bytes(payload);
58      }
59  
60      /**
61       * Constructs an instance of {@code ApnsNotification}.
62       *
63       * @param dtoken    The binary representation of the destination device token
64       * @param payload   The binary representation of the payload to be sent
65       */
66      public SimpleApnsNotification(byte[] dtoken, byte[] payload) {
67          this.deviceToken = Utilities.copyOf(dtoken);
68          this.payload = Utilities.copyOf(payload);
69      }
70  
71      /**
72       * Returns the binary representation of the device token.
73       *
74       */
75      public byte[] getDeviceToken() {
76          return Utilities.copyOf(deviceToken);
77      }
78  
79      /**
80       * Returns the binary representation of the payload.
81       *
82       */
83      public byte[] getPayload() {
84          return Utilities.copyOf(payload);
85      }
86  
87      private byte[] marshall = null;
88      /**
89       * Returns the binary representation of the message as expected by the
90       * APNS server.
91       *
92       * The returned array can be used to sent directly to the APNS server
93       * (on the wire/socket) without any modification.
94       */
95      public byte[] marshall() {
96          if (marshall == null)
97              marshall = Utilities.marshall(COMMAND, deviceToken, payload);
98          return marshall;
99      }
100 
101     /**
102      * Returns the length of the message in bytes as it is encoded on the wire.
103      *
104      * Apple require the message to be of length 255 bytes or less.
105      *
106      * @return length of encoded message in bytes
107      */
108     public int length() {
109         int length = 1 + 2 + deviceToken.length + 2 + payload.length;
110         assert marshall().length == length;
111         return length;
112     }
113 
114     @Override
115     public int hashCode() {
116         return 21
117                + 31 * Arrays.hashCode(deviceToken)
118                + 31 * Arrays.hashCode(payload);
119     }
120 
121     @Override
122     public boolean equals(Object obj) {
123         if (!(obj instanceof SimpleApnsNotification))
124             return false;
125         SimpleApnsNotification o = (SimpleApnsNotification)obj;
126         return Arrays.equals(this.deviceToken, o.deviceToken)
127                 && Arrays.equals(this.payload, o.payload);
128     }
129 
130     public int getIdentifier() {
131         return -1;
132     }
133 
134     public int getExpiry() {
135         return -1;
136     }
