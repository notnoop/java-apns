44  /***
45   * The class is used to create instances of {@link ApnsService}.
46   *
47   * Note that this class is not synchronized.  If multiple threads access a
48   * {@code ApnsServiceBuilder} instance concurrently, and at least on of the
49   * threads modifies one of the attributes structurally, it must be
50   * synchronized externally.
51   *
52   * Starting a new {@code ApnsService} is easy:
53   *
54   * <pre>
55   *   ApnsService = APNS.newService()
56   *    .withCert("/path/to/certificate.p12", "MyCertPassword")
57   *    .withSandboxDestination()
58   *    .build()
59   * </pre>
60   */
61  public class ApnsServiceBuilder {

63      private static final String KEY_ALGORITHM = "sunx509";

83      /***
84       * Specify the certificate used to connect to Apple APNS
85       * servers.  This relies on the path (absolute or relative to
86       * working path) to the keystore (*.p12) containing the
87       * certificate, along with the given password.
88       *
89       * The keystore needs to be of PKCS12 and the keystore
90       * needs to be encrypted using the SunX509 algorithm.  Both
91       * of these settings are the default.
92       *
93       * @param fileName  the path to the certificate
94       * @param password  the password of the keystore
95       * @return  this
96       */
97      public ApnsServiceBuilder withCert(String fileName, String password) {
98          FileInputStream stream = null;
99          try {
100             stream = new FileInputStream(fileName);
101             return withCert(stream, password);
102         } catch (FileNotFoundException e) {
103             throw new RuntimeException(e);
104         } finally {
105             Utilities.close(stream);
106         }
107     }

109     /***
110      * Specify the certificate used to connect to Apple APNS
111      * servers.  This relies on the stream of keystore (*.p12)
112      * containing the certificate, along with the given password.
113      *
114      * The keystore needs to be of PKCS12 and the keystore
115      * needs to be encrypted using the SunX509 algorithm.  Both
116      * of these settings are the default.
117      *
118      * @param stream    the keystore represented as input stream
119      * @param password  the password of the keystore
120      * @return  this
121      */
122     public ApnsServiceBuilder withCert(InputStream stream, String password) {
123         try {
124             return withSSLContext(
125                     newSSLContext(stream, password,
126                             KEYSTORE_TYPE, KEY_ALGORITHM));
127         } catch (Exception e) {
128             throw new RuntimeException(e);
129         }
130     }

269     /***
270      * Returns a fully initialized instance of {@link ApnsService},
271      * according to the requested settings.
272      *
273      * @return  a new instance of ApnsService
274      */
275     public ApnsService build() {
276         checkInitialization();
277         ApnsService service;
278 
279         SSLSocketFactory sslFactory = sslContext.getSocketFactory();
280         ApnsFeedbackConnection feedback = new ApnsFeedbackConnection(sslFactory, feedbackHost, feedbackPort);
281 
282         if (isNonBlocking) {
283             service = new MinaAdaptor(sslContext, gatewayHost, gatewaPort, feedback);
284         } else {
285             ApnsConnection conn = new ApnsConnectionImpl(sslFactory, gatewayHost, gatewaPort, forceReconnect);
286             if (pooledMax != 1) {
287                 conn = new ApnsPooledConnection(conn, pooledMax);
288             }
289 
290             service = new ApnsServiceImpl(conn, feedback);
291 
292             if (isQueued) {
293                 service = new QueuedApnsService(service);
294             }
295         }
296 
297         service.start();
298 
299         return service;
300     }
301 
302     private void checkInitialization() {
303         if (sslContext == null)
304             throw new IllegalStateException(
305                     "SSL Certificates and attribute are not initialized\n"
306                     + "Use .withCert() methods.");
307         if (gatewayHost == null || gatewaPort == -1)
308             throw new IllegalStateException(
309                     "The Destination APNS server is not stated\n"
310                     + "Use .withDestination(), withSandboxDestination(), "
311                     + "or withProductionDestination().");
312     }
