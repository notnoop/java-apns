49  /**
50   * The class is used to create instances of {@link ApnsService}.
51   *
52   * Note that this class is not synchronized.  If multiple threads access a
53   * {@code ApnsServiceBuilder} instance concurrently, and at least on of the
54   * threads modifies one of the attributes structurally, it must be
55   * synchronized externally.
56   *
57   * Starting a new {@code ApnsService} is easy:
58   *
59   * <pre>
60   *   ApnsService = APNS.newService()
61   *    .withCert("/path/to/certificate.p12", "MyCertPassword")
62   *    .withSandboxDestination()

91      /**
92       * Specify the certificate used to connect to Apple APNS
93       * servers.  This relies on the path (absolute or relative to
94       * working path) to the keystore (*.p12) containing the
95       * certificate, along with the given password.
96       *
97       * The keystore needs to be of PKCS12 and the keystore
98       * needs to be encrypted using the SunX509 algorithm.  Both
99       * of these settings are the default.
100      *
101      * @param fileName  the path to the certificate
102      * @param password  the password of the keystore
103      * @return  this
104      * @throws RuntimeIOException if it {@code fileName} cannot be
105      *          found or read
106      * @throws InvalidSSLConfig if fileName is invalid Keystore
107      *  or the password is invalid
108      */
109     public ApnsServiceBuilder withCert(String fileName, String password)
110     throws RuntimeIOException, InvalidSSLConfig {
111         FileInputStream stream = null;
112         try {
113             stream = new FileInputStream(fileName);
114             return withCert(stream, password);
115         } catch (FileNotFoundException e) {
116             throw new RuntimeIOException(e);
117         } finally {
118             Utilities.close(stream);
119         }
120     }

359     /**
360      * Returns a fully initialized instance of {@link ApnsService},
361      * according to the requested settings.
362      *
363      * @return  a new instance of ApnsService
364      */
365     public ApnsService build() {
366         checkInitialization();
367         ApnsService service;
368 
369         SSLSocketFactory sslFactory = sslContext.getSocketFactory();
370         ApnsFeedbackConnection feedback = new ApnsFeedbackConnection(sslFactory, feedbackHost, feedbackPort);
371 
372         if (isNonBlocking) {
373             service = new MinaAdaptor(sslContext, gatewayHost, gatewaPort, feedback);
374         } else {
375             ApnsConnection conn = new ApnsConnectionImpl(sslFactory, gatewayHost, gatewaPort, proxySocket, reconnectPolicy, delegate, errorDetection);
376             if (pooledMax != 1) {
377                 conn = new ApnsPooledConnection(conn, pooledMax);
378             }
379 
380             service = new ApnsServiceImpl(conn, feedback);
381 
382             if (isQueued) {
383                 service = new QueuedApnsService(service);
384             }
385         }
386 
387         service.start();
388 
389         return service;
390     }
391 
392     private void checkInitialization() {
393         if (sslContext == null)
394             throw new IllegalStateException(
395                     "SSL Certificates and attribute are not initialized\n"
396                     + "Use .withCert() methods.");
397         if (gatewayHost == null || gatewaPort == -1)
398             throw new IllegalStateException(
399                     "The Destination APNS server is not stated\n"
400                     + "Use .withDestination(), withSandboxDestination(), "
401                     + "or withProductionDestination().");
402     }
