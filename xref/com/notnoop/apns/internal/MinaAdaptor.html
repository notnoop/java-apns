33      private NioSocketConnector createNioSocketConnector(SSLContext sslContext) {
34          NioSocketConnector connector = new NioSocketConnector();
35          connector.setConnectTimeoutMillis(30 * 1000L);
36          connector.setHandler(new IoHandlerAdapter());
37          SslFilter sslFilter = new SslFilter(sslContext);
38          sslFilter.setUseClientMode(true);
39          connector.getFilterChain().addLast("SSL", sslFilter);
40          return connector;
41      }

43      @Override
44      public void push(ApnsNotification message) {
45          byte[] msg = message.marshall();
46          IoBuffer buf = IoBuffer.allocate(msg.length);
47          System.arraycopy(msg, 0, buf.array(), 0, msg.length);
48          cf.getSession().write(buf);
49      }

51      public void start() {
52          cf = connector.connect(new InetSocketAddress(host, port));
