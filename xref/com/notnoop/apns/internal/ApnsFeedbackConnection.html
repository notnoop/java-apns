31  package com.notnoop.apns.internal;
32  
33  import java.io.IOException;
34  import java.io.InputStream;
35  import java.net.Socket;
36  import java.util.Date;
37  import java.util.Map;
38  
39  import javax.net.SocketFactory;
40  
41  import org.slf4j.Logger;
42  import org.slf4j.LoggerFactory;
43  
44  import com.notnoop.exceptions.NetworkIOException;
45  
46  public class ApnsFeedbackConnection {
47      private static final Logger logger = LoggerFactory.getLogger(ApnsFeedbackConnection.class);
48  
49      private final SocketFactory factory;
50      private final String host;
51      private final int port;
52  
53      public ApnsFeedbackConnection(SocketFactory factory, String host, int port) {
54          this.factory = factory;
55          this.host = host;
56          this.port = port;
57      }
58  
59      int DELAY_IN_MS = 1000;
60      private static final int RETRIES = 3;
61  
62      public Map<String, Date> getInactiveDevices() throws NetworkIOException {
63          int attempts = 0;
64          while (true) {
65              try {
66                  attempts++;
67                  Map<String, Date> result = getInactiveDevicesImpl();
68  
69                  attempts = 0;
70                  return result;
71              } catch (Exception e) {
72                  logger.warn("Failed to retreive invalid devices", e);
73                  if (attempts >= RETRIES) {
74                      logger.error("Couldn't get feedback connection", e);
75                      Utilities.wrapAndThrowAsRuntimeException(e);
76                  }
77                  Utilities.sleep(DELAY_IN_MS);
78              }
79          }
80      }
81  
82      public Map<String, Date> getInactiveDevicesImpl() throws IOException {
83          Socket socket = null;
84          try {
85              socket = factory.createSocket(host, port);
86              InputStream stream = socket.getInputStream();
87              return Utilities.parseFeedbackStream(stream);
88          } finally {
89              Utilities.close(socket);
90          }
91      }
92  
93  }
