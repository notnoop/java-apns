31  package com.notnoop.apns.internal;

33  import java.io.IOException;
34  import java.net.Socket;
35  
36  import javax.net.SocketFactory;
37  
38  import org.slf4j.Logger;
39  import org.slf4j.LoggerFactory;
40  
41  import com.notnoop.apns.ApnsNotification;
42  
43  public class ApnsConnectionImpl implements ApnsConnection {
44      private static final Logger logger = LoggerFactory.getLogger(ApnsConnectionImpl.class);
45  
46      private final SocketFactory factory;
47      private final String host;
48      private final int port;
49      private final boolean forceReconnect;
50  
51      public ApnsConnectionImpl(SocketFactory factory, String host, int port) {
52          this(factory, host, port, false);
53      }
54  
55      public ApnsConnectionImpl(SocketFactory factory, String host, int port, boolean forceReconnect) {
56          this.factory = factory;
57          this.host = host;
58          this.port = port;
59          this.forceReconnect = forceReconnect;
60      }
61  
62      public synchronized void close() {
63          try {
64              if (socket != null)
65                  this.socket.close();
66          } catch (IOException e) {
67              logger.debug("Error while closing socket connection", e);
68          }
69      }
70  
71      private Socket socket;
72      private synchronized Socket socket() {
73          if (forceReconnect) {
74              Utilities.close(socket);
75              socket = null;
76          }
77  
78          while (socket == null || socket.isClosed()) {
79              try {
80                  socket = factory.createSocket(host, port);
81                  logger.debug("Made a new connection to APNS");
82              } catch (Exception e) {
83                  logger.error("Couldnt' connec to APNS server", e);
84              }
85          }
86          return socket;
87      }
88  
89      int DELAY_IN_MS = 1000;
90  
91      private static final int RETRIES = 3;
92      public synchronized void sendMessage(ApnsNotification m) {
93          int attempts = 0;
94          while (true) {
95              try {
96                  attempts++;
97                  Socket socket = socket();
98                  socket.getOutputStream().write(m.marshall());
99                  socket.getOutputStream().flush();
100                 logger.debug("Message \"{}\" sent", m);
101 
102                 attempts = 0;
103                 break;
104             } catch (Exception e) {
105                 if (attempts >= RETRIES) {
106                     logger.error("Couldn't send message " + m, e);
107                     throw new RuntimeException(e);
108                 }
109                 logger.warn("Failed to send message " + m + "... trying again", e);
110                 Utilities.sleep(DELAY_IN_MS);
111                 Utilities.close(socket);
112                 socket = null;
113             }
114         }
115     }
116 
117     public ApnsConnectionImpl copy() {
118         logger.debug("New Copy!");
119         return new ApnsConnectionImpl(factory, host, port, forceReconnect);
120     }
