31  package com.notnoop.apns;
32  
33  import java.util.Arrays;
34  
35  import com.notnoop.apns.internal.Utilities;
36  
37  /***
38   * Represents an APNS notification to be sent to Apple service.
39   */
40  public class ApnsNotification {
41  
42      private final static byte COMMAND = 0;
43      private final byte[] deviceToken;
44      private final byte[] payload;
45  
46      /***
47       * Constructs an instance of {@code ApnsNotification}.
48       *
49       * The message encodes the payload with a {@code UTF-8} encoding.
50       *
51       * @param dtoken    The Hex of the device token of the destination phone
52       * @param payload   The payload message to be sent
53       */
54      public ApnsNotification(String dtoken, String payload) {
55          this.deviceToken = Utilities.decodeHex(dtoken);
56          this.payload = Utilities.toUTF8Bytes(payload);
57      }
58  
59      /***
60       * Constructs an instance of {@code ApnsNotification}.
61       *
62       * @param dtoken    The binary representation of the destination device token
63       * @param payload   The binary representation of the payload to be sent
64       */
65      public ApnsNotification(byte[] dtoken, byte[] payload) {
66          this.deviceToken = Utilities.copyOf(dtoken);
67          this.payload = Utilities.copyOf(payload);
68      }
69  
70      /***
71       * Returns the binary representation of the device token.
72       *
73       */
74      public byte[] getDeviceToken() {
75          return Utilities.copyOf(deviceToken);
76      }
77  
78      /***
79       * Returns the binary representation of the payload.
80       *
81       */
82      public byte[] getPayload() {
83          return Utilities.copyOf(payload);
84      }
85  
86      private byte[] marshall = null;
87      /***
88       * Returns the binary representation of the message as expected by the
89       * APNS server.
90       *
91       * The returned array can be used to sent directly to the APNS server
92       * (on the wire/socket) without any modification.
93       */
94      public byte[] marshall() {
95          if (marshall == null)
96              marshall = Utilities.marshall(COMMAND, deviceToken, payload);
97          return marshall;
98      }
99  
100     /***
101      * Returns the length of the message in bytes as it is encoded on the wire.
102      *
103      * Apple require the message to be of length 255 bytes or less.
104      *
105      * @return length of encoded message in bytes
106      */
107     public int length() {
108         int length = 1 + 2 + deviceToken.length + 2 + payload.length;
109         assert marshall().length == length;
110         return length;
111     }
112 
113     @Override
114     public int hashCode() {
115         return 21
116                + 31 * Arrays.hashCode(deviceToken)
117                + 31 * Arrays.hashCode(payload);
118     }
119 
120     @Override
121     public boolean equals(Object obj) {
122         if (!(obj instanceof ApnsNotification))
123             return false;
124         ApnsNotification o = (ApnsNotification)obj;
125         return Arrays.equals(this.deviceToken, o.deviceToken)
126                 && Arrays.equals(this.payload, o.payload);
127     }
128 }
