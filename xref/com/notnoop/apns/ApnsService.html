31  package com.notnoop.apns;
32  
33  import java.util.Collection;
34  import java.util.Date;
35  import java.util.Map;
36  
37  /***
38   * Represents the connection and interface to the Apple APNS servers.
39   *
40   * The service is created by {@link ApnsServiceBuilder} like:
41   *
42   * <pre>
43   *   ApnsService = APNS.newService()
44   *                  .withCert("/path/to/certificate.p12", "MyCertPassword")
45   *                  .withSandboxDestination()
46   *                  .build()
47   * </pre>
48   */
49  public interface ApnsService {
50  
51      /***
52       * Sends a push notification with the provided {@code payload} to the
53       * iPhone of {@code deviceToken}.
54       *
55       * The payload needs to be a valid JSON object, otherwise it may fail
56       * silently.  It is recommended to use {@link PayloadBuilder} to create
57       * one.
58       *
59       * @param deviceToken   the destination iPhone device token
60       * @param payload       The payload message
61       */
62      void push(String deviceToken, String payload);
63  
64      /***
65       * Sends a push notification with the provided {@code payload} to the
66       * iPhone of {@code deviceToken}.
67       *
68       * The payload needs to be a valid JSON object, otherwise it may fail
69       * silently.  It is recommended to use {@link PayloadBuilder} to create
70       * one.
71       *
72       * @param deviceToken   the destination iPhone device token
73       * @param payload       The payload message
74       */
75      void push(byte[] deviceToken, byte[] payload);
76  
77      /***
78       * Sends a bulk push notification with the provided
79       * {@code payload} to iPhone of {@code deviceToken}s set.
80       *
81       * The payload needs to be a valid JSON object, otherwise it may fail
82       * silently.  It is recommended to use {@link PayloadBuilder} to create
83       * one.
84       *
85       * @param deviceTokens   the destination iPhone device tokens
86       * @param payload       The payload message
87       */
88      void push(Collection<String> deviceTokens, String payload);
89  
90      /***
91       * Sends a bulk push notification with the provided
92       * {@code payload} to iPhone of {@code deviceToken}s set.
93       *
94       * The payload needs to be a valid JSON object, otherwise it may fail
95       * silently.  It is recommended to use {@link PayloadBuilder} to create
96       * one.
97       *
98       * @param deviceTokens   the destination iPhone device tokens
99       * @param payload       The payload message
100      */
101     void push(Collection<byte[]> deviceTokens, byte[] payload);
102 
103     /***
104      * Sends the provided notification {@code message} to the desired
105      * destination.
106      */
107     void push(ApnsNotification message);
108 
109     /***
110      * Starts the service.
111      *
112      * The underlying implementation may prepare its connections or
113      * datastructures to be able to send the messages.
114      *
115      * This method is a blocking call, even if the service represents
116      * a Non-blocking push service.  Once the service is returned, it is ready
117      * to accept push requests.
118      */
119     void start();
120 
121     /***
122      * Stops the service and frees any allocated resources it created for this
123      * service.
124      *
125      * The underlying implementation should close all connections it created,
126      * and possibly stop any threads as well.
127      */
128     void stop();
129 
130     /***
131      * Returns the list of devices that reported failed-delivery
132      * attempts to the Apple Feedback services.
133      *
134      * The result is map, mapping the device tokens as Hex Strings
135      * mapped to the timestamp when APNs determined that the
136      * application no longer exists on the device.
137      */
138     Map<String, Date> getInactiveDevices();
139 }
