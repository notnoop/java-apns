41  /**
42   * Signals that the the provided SSL context settings (e.g.
43   * keystore path, password, encryption type, etc) are invalid
44   *
45   * This Exception can be caused by any of the following:
46   *
47   * <ol>
48   * <li>{@link KeyStoreException}</li>
49   * <li>{@link NoSuchAlgorithmException}</li>
50   * <li>{@link CertificateException}</li>
51   * <li>{@link IOException}</li>
52   * <li>{@link UnrecoverableKeyException}</li>
53   * <li>{@link KeyManagementException}</li>
54   * </ol>
55   *
56   */
57  public class InvalidSSLConfig extends ApnsException {
58      private static final long serialVersionUID = -7283168775864517167L;
59  
60      public InvalidSSLConfig()                      { super(); }
61      public InvalidSSLConfig(String message)        { super(message); }
62      public InvalidSSLConfig(Throwable cause)       { super(cause); }
63      public InvalidSSLConfig(String m, Throwable c) { super(m, c); }
