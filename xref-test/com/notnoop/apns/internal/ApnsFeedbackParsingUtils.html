14      static byte[] pack(byte[]... args) {
15          int total = 0;
16          for (byte[] arg : args)
17              total += arg.length;
18  
19          byte[] result = new byte[total];
20  
21          int index = 0;
22          for (byte[] arg : args) {
23              System.arraycopy(arg, 0, result, index, arg.length);
24              index += arg.length;
25          }
26          return result;
27      }
